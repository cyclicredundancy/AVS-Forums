By the way, do you really think that this is SIMPLE?????

ox=-1;
oy=1;
oz=0;

movx=-0.9;
movy=-0.3;
movz=-0.5;

x1=x;
y1=y;
z1=1;

cmovz=cos(movz);
smovz=sin(movz);
cmovy=cos(movy);
smovy=sin(movy);
cmovx=cos(movx);
smovx=sin(movx);

//rotation
x2=x1*cmovz-y1*smovz;
y2=x1*smovz+y1*cmovz;

x1=x2*cmovy+z1*smovy;
z2=-x2*smovy+z1*cmovy;

y1=y2*cmovx-z2*smovx;
z1=y2*smovx+z2*cmovx;

//tube
a=2*(sqr(x1)+sqr(y1));
b=2*(ox*x1+oy*y1);
c=2*(sqr(ox)+sqr(oy)-1);

t1=sqr(b)-a*c;
t=if(below(t1,0),-10,sqrt(t1));
t=(-b-t)/a;

//plane
flat=if(below(oy,0),10,0-oy/y1);

//intersect
a=min(flat,t);
chk=equal(flat,a);

//offset
x1=x1*a+ox;
y1=y1*a+oy;
z1=z1*a+oz;

//project
x=z1;
y=if(chk,x1*0.2-0.4,asin(sin(atan2(y1,x1))))+1.3;
